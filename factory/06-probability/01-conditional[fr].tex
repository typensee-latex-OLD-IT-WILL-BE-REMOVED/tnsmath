% ---------------------- %


\subsection{Probabilité conditionnelle}

\paragraph{Un exemple type}

La 1\iere{} notation, qui est devenue standard, permet de comprendre l'ordre des arguments.
\begin{latexex}
$\probacond{B}{A}
 =
 \probacond*{B}{A}
 =
 \probacond**{B}{A}
 =
 \dprobacond**{B}{A}$
\end{latexex}


% ---------------------- %


\paragraph{Choisir le nom de la probabilité}

\begin{latexex}
$\probacond[P]{B}{A}
 =
 \probacond*[P]{B}{A}
 =
 \probacond**[P]{B}{A}
 =
 \dprobacond**[P]{B}{A}$
\end{latexex}


% ---------------------- %


\subsubsection{Fiches techniques}

\paragraph{Probabilité conditionnelle}

\IDmacro{probacond}{1}{2}

\IDmacro{probacond*}{1}{2}

\IDmacro{probacond**}{1}{2}

\IDmacro{dprobacond**}{1}{2}

\IDoption{} le nom de la probabilité.

\IDarg{1} l'ensemble qui donne la condition.

\IDarg{2} l'ensemble dont on veut calculer la probabilité.
