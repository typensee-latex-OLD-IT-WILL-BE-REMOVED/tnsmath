\subsection{Norme}

\begin{latexex}
 $\norm{\vect{i}} = \vnorm{i}$

 $\norm {\dfrac{2}{7} \vect*{e}{k}}
= \norm*{\dfrac{2}{7} \vect*{e}{k}}$
\end{latexex}
