% separator: |; before: \begin{pmatrix}; between: \\, after: \end{pmatrix}
\newcommand{\vcoord}[1]{%
  \makemultiargument{|}{#1}{\begin{pmatrix}}{\\}{\end{pmatrix}}%
}

% separator: +; before: ((; between: -, after: ))
\newcommand{\minify}[1]{%
  \makemultiargument{+}{#1}{((}{-}{))}%
}

\NewDocumentCommand{\boxify}{m}{%
  \makemultiargument{+}{#1}{(\,}{-}{\,)}[\fbox{##1}]%
}
